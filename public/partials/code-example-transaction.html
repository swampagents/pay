// Initialize SWAMP PAY client
import { SwampPay } from '@swamp/sdk';

const client = new SwampPay({
  terminalId: 'SP-700-001',
  encryptionKey: process.env.SWAMP_KEY,
});

// Process an NFC payment
const tx = await client.charge({
  amount: 4200,
  currency: 'USD',
  method: 'nfc_tap',
});

// Listen for settlement confirmation
client.on('charge.settled', (event) => {
  console.log(event.id);  // tx_7f8a9b...
  console.log(event.data_retained);  // false
});
